### 快速开始

创建三个文件

webpack.common.js

webpack.dev.js

webpack.prod.js


```js
// package.json
"script": {
  build:prod: "webpack --config ./config/webpack.common.js --env production",
  build:dev: "webpack --config ./config/webpack.common.js --env development"
}
```


```js
// webpack.common.js

const { merge } = require('webpack-merge')  // 以前版本直接导入
const prodConfig = require('./webpack.prod.js')
const devConfig = require('./webpack.dev.js')
module.export = (env) => {
    const isProd = env.production
    // 这个env 就是通过 package.json 里面 --env 传进来的
    const baseConfig = {
      resolve: {
      '@': resolveApp('./src')  // 定位到我们应用根目录下下面的 src 目录
    }
    output: {
      path: resolveApp('./dist')
    }
  }
  const config = isProd ? prodConfig : devConfig
  return merge(baseConfig, config)
}
```


```js
// paths.js

// 找到当前的根目录，就是从哪块开始打包
const path = require('path');

const appDir = process.cwd();

const resolveApp = (relativePath) => {
  return path.resolve(appDir, relativePath)
}

module.exports = resolveApp
```
